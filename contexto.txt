class QueryAgent:
    """
    Responsabilidades:
    - Intent Classification: búsqueda, comparación, compra, soporte, información
    - Entity Extraction: categorías, precios, marcas, especificaciones, ubicación
    - Context Enrichment: historial + preferencias + sesión actual
    - Query Expansion: sinónimos, términos relacionados, búsquedas similares
    """
    
    async def process_query(self, message: str, user_context: dict) -> QueryIntent:
        # 1. Clasificar intención principal
        intent = await self.classify_intent(message)
        
        # 2. Extraer entidades nombradas
        entities = await self.extract_entities(message)
        
        # 3. Expandir consulta con contexto
        expanded_query = await self.expand_with_context(message, user_context)
        
        # 4. Generar parámetros de búsqueda
        search_params = await self.build_search_params(intent, entities)
        
        return QueryIntent(
            original_message=message,
            intent_type=intent,
            entities=entities,
            expanded_terms=expanded_query,
            search_parameters=search_params,
            confidence_score=self.calculate_confidence()
        )

class ActionAgent:
    """
    Responsabilidades:
    - Cart Management: CRUD operations on cart
    - Order Creation: multi-vendor order processing
    - Vendor Communication: stock checks, custom requests
    - Payment Processing: integration con payment gateways
    - User Preference Learning: update ML models
    """
    
    async def execute_action(self, action_intent: ActionIntent) -> ActionResult:
        match action_intent.action_type:
            case "add_to_cart":
                return await self.add_to_cart(action_intent)
            case "create_order":
                return await self.create_multivendor_order(action_intent)
            case "compare_products":
                return await self.compare_products(action_intent)
            case "set_price_alert":
                return await self.create_price_alert(action_intent)
            case "contact_vendor":
                return await self.contact_vendor(action_intent)
    
    async def create_multivendor_order(self, intent: ActionIntent) -> OrderResult:
        # 1. Group cart items by vendor
        vendor_groups = self.group_items_by_vendor(intent.cart_items)
        
        # 2. Verify stock availability
        stock_verification = await self.verify_stock_all_vendors(vendor_groups)
        
        # 3. Calculate vendor splits and commissions
        financial_splits = self.calculate_vendor_splits(vendor_groups)
        
        # 4. Create order via Django API
        order_data = await self.django_client.create_order({
            "user_id": intent.user_id,
            "items": intent.cart_items,
            "vendor_splits": financial_splits,
            "ai_conversation_id": intent.conversation_id
        })
        
        # 5. Trigger fulfillment workflows
        await self.trigger_vendor_notifications(vendor_groups)
        
        return OrderResult(success=True, order=order_data)

### **Orquestador Central**
```python
class AIOrchestrator:
    """
    Coordina la comunicación entre agentes y maneja el flujo completo
    de una conversación de chat hasta la acción final.
    """
    
    def __init__(self):
        self.query_agent = QueryAgent()
        self.rag_agent = RAGAgent()
        self.action_agent = ActionAgent()
        self.product_agent = ProductAgent()
        self.analytics_agent = AnalyticsAgent()
    
    async def process_message(self, chat_message: ChatMessage) -> ChatResponse:
        # 1. Parse y entender la consulta
        query_intent = await self.query_agent.process_query(
            chat_message.message, 
            chat_message.context
        )
        
        # 2. Ejecutar búsqueda si es necesario
        if query_intent.requires_search:
            search_results = await self.rag_agent.search_products(query_intent)
        
        # 3. Ejecutar acciones si es necesario
        if query_intent.requires_action:
            action_result = await self.action_agent.execute_action(query_intent)
        
        # 4. Generar respuesta conversacional
        response = await self.generate_conversational_response(
            query_intent, search_results, action_result
        )
        
        # 5. Track interaction para aprendizaje
        await self.analytics_agent.track_interaction(chat_message, response)
        
        return ChatResponse(
            message=response.text,
            products=search_results.products if search_results else [],
            actions_taken=action_result.actions if action_result else [],
            suggested_followups=response.followup_suggestions
        )

        await self.session.patch(
            f"{self.base_url}/api/products/{product_id}/analytics/",
            json=analytics
        )

### **Data Flow - Proceso de Embedding**

```python
# Background Process: Generar embeddings para productos nuevos
class EmbeddingPipeline:
    async def process_new_products(self):
        # 1. Obtener productos sin embedding de Django
        products = await self.django_client.get_products_without_embeddings()
        
        for product in products:
            # 2. Preparar texto para embedding
            text_content = self.prepare_product_text(product)
            
            # 3. Generar embedding con OpenAI
            embedding = await self.openai_client.embeddings.create(
                model="text-embedding-3-large",
                input=text_content
            )
            
            # 4. Almacenar en ChromaDB
            await self.chroma_client.add(
                documents=[text_content],
                embeddings=[embedding.data[0].embedding],
                metadatas=[{
                    "product_id": product["id"],
                    "vendor_id": product["vendor_id"],
                    "category": product["category"],
                    "price": product["price"]
                }],
                ids=[f"product_{product['id']}"]
            )
            
            # 5. Marcar como procesado en Django
            await self.django_client.mark_embedding_generated(product["id"])
    
    def prepare_product_text(self, product: Dict) -> str:
        return f"""
        Producto: {product['name']}
        Categoría: {product['category']}
        Descripción: {product['description']}
        Especificaciones: {product.get('specifications', {})}
        Precio: ${product['price']}
        Vendor: {product['vendor_name']}
        Tags: {', '.join(product.get('tags', []))}
        """.strip()

## 🛠️ TOOLS Y COMANDOS DE DESARROLLO

### **Makefile para Comandos Frecuentes**

```makefile
# Makefile
.PHONY: help build up down restart logs shell test clean

help: ## Mostrar ayuda
	@grep -E '^[a-zA-Z_-]+:.*?## .*$$' $(MAKEFILE_LIST) | sort | awk 'BEGIN {FS = ":.*?## "}; {printf "\033[36m%-30s\033[0m %s\n", $$1, $$2}'

build: ## Construir todos los containers
	docker-compose build

up: ## Levantar todos los servicios
	docker-compose up -d

down: ## Bajar todos los servicios
	docker-compose down

restart: ## Reiniciar todos los servicios
	docker-compose restart

logs: ## Ver logs de todos los servicios
	docker-compose logs -f

shell-django: ## Shell en Django container
	docker-compose exec django python manage.py shell

shell-ai: ## Shell en AI Engine container
	docker-compose exec ai-engine python

test-django: ## Ejecutar tests Django
	docker-compose exec django python manage.py test

test-ai: ## Ejecutar tests FastAPI
	docker-compose exec ai-engine pytest

migrate: ## Ejecutar migraciones Django
	docker-compose exec django python manage.py migrate

makemigrations: ## Crear migraciones Django
	docker-compose exec django python manage.py makemigrations

collectstatic: ## Recopilar archivos estáticos
	docker-compose exec django python manage.py collectstatic --noinput

loaddata: ## Cargar datos de prueba
	docker-compose exec django python manage.py loaddata database/seeds/*.json

clean: ## Limpiar containers y volúmenes
	docker-compose down -v
	docker system prune -f

setup: ## Setup inicial completo
	cp .env.example .env
	docker-compose up -d postgres redis
	sleep 10
	docker-compose up -d django
	sleep 10
	make migrate
	make loaddata
	docker-compose up -d

dev: ## Modo desarrollo
	docker-compose -f docker-compose.yml -f docker-compose.override.yml up
```
